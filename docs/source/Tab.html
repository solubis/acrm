app.views.Tab = Ext.extend(Ext.Panel, {
	
	iconCls: 'favorites',
    title: 'Today',

	listeners: {
		scope: this,
		activate: function(tab) {
			tab.add(tab.innerView);
			tab.doLayout();
		},
		deactivate: function(tab) {
			tab.removeAll(true);
		}
	},
	
	setInnerView: function(innerView){
		this.innerView = innerView;
	}
});
